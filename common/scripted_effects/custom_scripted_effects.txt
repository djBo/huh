huh_neut_character = {
	add_character_modifier = {
		name = will_not_marry
		duration = -1
		hidden = yes
	}
	add_trait = cannot_marry
	add_trait = celibate
	if = {
		limit = { is_female = yes }
		add_trait = nun
	}
	if = {
		limit = { is_female = no}
		add_trait = monk
	}
}

huh_marry_spouse = {
	set_character_flag = is_huh_spouse
	if = {
		limit = { is_female = yes }
		add_spouse = FROM
	}
	if = {
		limit = { is_female = no }
		add_spouse_matrilineal = FROM
	}
	remove_trait = homosexual
	remove_trait = chaste
	change_intrigue = -1
	add_trait = lustful
	change_learning = 1
}

huh_fire_all = {
	# Fire all commanders
	any_courtier_or_vassal = {
		limit = {
			has_minor_title = title_commander
		}
		log = "Commander [This.GetBestName]"
		remove_title = title_commander
	}
	# Fire counsil
	job_chancellor = {
		log = "Chancellor [This.GetBestName]"
		remove_title = job_chancellor
	}
	job_marshal = {
		log = "Marshal [This.GetBestName]"
		remove_title = job_marshal
	}
	job_treasurer = {
		log = "Treasurer [This.GetBestName]"
		remove_title = job_treasurer
	}
	job_spymaster = {
		log = "Spymaster [This.GetBestName]"
		remove_title = job_spymaster
	}
	job_spiritual = {
		log = "Spiritual [This.GetBestName]"
		remove_title = job_spiritual
	}
	# Fire advisor 1
	any_courtier_or_vassal = {
		limit = {
			has_minor_title = title_councilmember_king
		}
		log = "Advisor I [This.GetBestName]"
		remove_title = title_councilmember_king
	}
	# Fire advisor 2
	any_courtier_or_vassal = {
		limit = {
			has_minor_title = title_councilmember_emperor
		}
		log = "Advisor II [This.GetBestName]"
		remove_title = title_councilmember_emperor
	}
}
